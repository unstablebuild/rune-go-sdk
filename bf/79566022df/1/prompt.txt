// setupE2EExecutor creates a sandbox executor for e2e testing.
// Uses the mock runtime for now; real e2e tests will use actual VMs.
func setupE2EExecutor(t *testing.T, config Config) *Executor {
    t.Helper()

    if config.WorkspaceRoot == "" {
        config.WorkspaceRoot = "/tmp"
    }

    exec, err := New(context.Background(), config, WithRuntime(newMockRuntime()))
    require.NoError(t, err)
    t.Cleanup(func() { _ = exec.Close() })

    return exec
}

func runCommand(t *testing.T, exec *Executor, command string) (string, string) {
    t.Helper()

    stdout := &bytes.Buffer{}
    stderr := &bytes.Buffer{}
    watchCh := make(chan error, 1)

    cmd := workspaceapi.Cmd{
        Path:    "/bin/sh",
        Args:    []string{"-c", command},
        Stdout:  stdout,
        Stderr:  stderr,
        Watcher: &testWatcher{ch: watchCh},
    }

    ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
    defer cancel()

    _, err := exec.Start(ctx, cmd)
    require.NoError(t, err)

    select {
    case <-watchCh:
    case <-ctx.Done():
        t.Fatalf("timeout waiting for command: %s", command)
    }

    return stdout.String(), stderr.String()
}

func TestE2E_SimpleCommand(t *testing.T) {
    exec := setupE2EExecutor(t, DefaultConfig())

    stdout, _ := runCommand(t, exec, "echo hello")
    assert.Equal(t, "hello\n", stdout)
}

func TestE2E_EnvironmentVariables(t *testing.T) {
    exec := setupE2EExecutor(t, DefaultConfig())

    tsuite := []struct {
        name    string
        cmd     string
        wantOut string
    }{
        {
            name:    "echo with var",
            cmd:     "export FOO=bar && echo $FOO",
            wantOut: "bar\n",
        },
        {
            name:    "multiple vars",
            cmd:     "export A=1 B=2 && echo $A$B",
            wantOut: "12\n",
        },
    }

    for _, tc := range tsuite {
        t.Run(tc.name, func(t *testing.T) {
            stdout, _ := runCommand(t, exec, tc.cmd)
            assert.Equal(t, tc.wantOut, stdout)
        })
    }
}

func TestE2E_WorkingDirectory(t *testing.T) {
    exec := setupE2EExecutor(t, DefaultConfig())

    stdout, _ := runCommand(t, exec, "cd /tmp && pwd")
    // On macOS /tmp is symlinked to /private/tmp
    assert.Contains(t, stdout, "tmp")
}

func TestE2E_StderrCapture(t *testing.T) {
    exec := setupE2EExecutor(t, DefaultConfig())

    stdout, stderr := runCommand(t, exec, "echo out; echo err >&2")
    assert.Equal(t, "out\n", stdout)
    assert.Equal(t, "err\n", stderr)
}

func TestE2E_ExitCode(t *testing.T) {
    exec := setupE2EExecutor(t, DefaultConfig())

    watchCh := make(chan error, 1)
    cmd := workspaceapi.Cmd{
        Path:    "/bin/sh",
        Args:    []string{"-c", "exit 42"},
        Watcher: &testWatcher{ch: watchCh},
    }

    ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
    defer cancel()

    _, err := exec.Start(ctx, cmd)
    require.NoError(t, err)

    select {
    case werr := <-watchCh:
        require.Error(t, werr)
        assert.Contains(t, werr.Error(), "42")
    case <-ctx.Done():
        t.Fatal("timeout")
    }
}
